    <!-- Tab Functionality Script -->
    <script>
      document.addEventListener('DOMContentLoaded', function() {
        // Tab functionality
        const tabs = document.querySelectorAll('.property-tab');
        const tabContents = document.querySelectorAll('.property-listings');
        
        tabs.forEach(tab => {
          tab.addEventListener('click', function() {
            // Remove active class from all tabs
            tabs.forEach(t => t.classList.remove('active'));
            // Add active class to current tab
            this.classList.add('active');
            
            // Get the tab id
            const tabId = this.dataset.tab;
            
            // Hide all tab contents
            tabContents.forEach(content => content.classList.remove('active'));
            
            // Show current tab content
            document.getElementById(tabId).classList.add('active');
          });
        });
        
        // Voice search functionality
        const voiceSearchButton = document.getElementById('voice-search-button');
        
        if (voiceSearchButton && 'SpeechRecognition' in window || 'webkitSpeechRecognition' in window) {
          const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
          const recognition = new SpeechRecognition();
          
          recognition.continuous = false;
          recognition.interimResults = false;
          recognition.lang = 'en-US';
          
          voiceSearchButton.addEventListener('click', function() {
            // Visual feedback that voice search is active
            this.innerHTML = '<i class="fas fa-microphone-alt"></i> Listening...';
            this.classList.add('listening');
            
            // Start recognition
            recognition.start();
          });
          
          recognition.onresult = function(event) {
            const transcript = event.results[0][0].transcript;
            console.log('Voice search query:', transcript);
            
            // Process the voice query
            processVoiceQuery(transcript);
            
            // Reset button
            voiceSearchButton.innerHTML = '<i class="fas fa-microphone"></i> Search Properties by Voice';
            voiceSearchButton.classList.remove('listening');
          };
          
          recognition.onerror = function(event) {
            console.error('Speech recognition error:', event.error);
            
            // Reset button
            voiceSearchButton.innerHTML = '<i class="fas fa-microphone"></i> Search Properties by Voice';
            voiceSearchButton.classList.remove('listening');
            
            // Show error message
            alert('Voice recognition error: ' + event.error);
          };
          
          recognition.onend = function() {
            // Reset button if recognition ends without result
            if (voiceSearchButton.classList.contains('listening')) {
              voiceSearchButton.innerHTML = '<i class="fas fa-microphone"></i> Search Properties by Voice';
              voiceSearchButton.classList.remove('listening');
            }
          };
        } else {
          // Hide voice search button if not supported
          if (voiceSearchButton) {
            voiceSearchButton.style.display = 'none';
          }
        }
        
        // Simple voice query processor
        function processVoiceQuery(query) {
          // Convert to lowercase for easier matching
          const lowerQuery = query.toLowerCase();
          
          // Redirect based on query keywords
          if (lowerQuery.includes('newest') || lowerQuery.includes('new listings') || lowerQuery.includes('recent listings')) {
            // Activate newest tab
            document.querySelector('.property-tab[data-tab="newest"]').click();
            scrollToSection('properties');
          } else if (lowerQuery.includes('price low') || lowerQuery.includes('affordable') || lowerQuery.includes('cheapest')) {
            // Activate price low tab
            document.querySelector('.property-tab[data-tab="price-low"]').click();
            scrollToSection('properties');
          } else if (lowerQuery.includes('price high') || lowerQuery.includes('expensive') || lowerQuery.includes('luxury')) {
            // Activate price high tab
            document.querySelector('.property-tab[data-tab="price-high"]').click();
            scrollToSection('properties');
          } else if (lowerQuery.includes('all properties') || lowerQuery.includes('show all')) {
            // Activate all properties tab
            document.querySelector('.property-tab[data-tab="all"]').click();
            scrollToSection('properties');
          } else if (lowerQuery.includes('sold') || lowerQuery.includes('recently sold')) {
            // Activate sold tab
            document.querySelector('.property-tab[data-tab="sold"]').click();
            scrollToSection('properties');
          } else if (lowerQuery.includes('contact') || lowerQuery.includes('get in touch')) {
            // Scroll to contact section
            scrollToSection('contact');
          } else if (lowerQuery.includes('neighborhood') || lowerQuery.includes('area') || lowerQuery.includes('community')) {
            // Scroll to neighborhood section
            scrollToSection('neighborhood');
          } else if (lowerQuery.includes('faq') || lowerQuery.includes('questions')) {
            // Scroll to FAQ section
            scrollToSection('faq');
          } else {
            // Default to showing all properties
            document.querySelector('.property-tab[data-tab="newest"]').click();
            scrollToSection('properties');
          }
        }
        
        // Helper function to scroll to section
        function scrollToSection(sectionId) {
          const section = document.getElementById(sectionId);
          if (section) {
            section.scrollIntoView({ behavior: 'smooth' });
          }
        }
      });
    </script>
